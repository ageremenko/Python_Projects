# обрабатываем пропущенные значения
for i in main_df.drop(columns=current_status):
    unique_values = main_df[i].unique()
    
    # находим и заполняем неуказанные категориальные атрибуты значениями -1
    if (main_df[i].max()-main_df[i].min())==(len(unique_values)-2):
        main_df[i].fillna(main_df[i].min()-1, inplace=True)
        cat_attr.append(main_df[i].name)
    
    # находим и заполняем потерянные категориальные атрибуты,
    # т.е. те, у которых значения последовательны и видно, что не хватает категории
    elif ((main_df[i].max()-main_df[i].min())==(len(unique_values)-1)):
        for j in range(int(main_df[i].min()), int(main_df[i].max()+1)):
            if j not in unique_values:
                main_df[i].fillna(j, inplace=True)
                cat_attr.append(main_df[i].name)
    
    # для всех остальных, считаем, что они числовые, а значит отсутствие значения - отсутствие признака
    else:
        main_df[i].fillna(0, inplace=True)
        num_attr.append(main_df[i].name)
